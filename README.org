** Select the version of the docker image

#+BEGIN_SRC sh
IMAGE=couchdb
TAG=3.1.1
#+END_SRC

** Set container name

#+BEGIN_SRC sh
NAME=couchdb
#+END_SRC

** Pull the docker image

#+BEGIN_SRC sh
docker pull $IMAGE:$TAG
#+END_SRC

** Configure CouchDB

Contents of the .env file.

#+BEGIN_SRC conf :tangle .env.dist
COUCHDB_USER=user
COUCHDB_PASSWORD=secretpassword
#+END_SRC
